/*s
 * bitXor - x^y using only ~ and &
 *   Example: bitXor(4, 5) = 1
 *   Legal ops: ~ &
 *   Max ops: 14
 *   Rating: 1
 */
int bitXor(int x, int y) {
  // 德摩根定律
  // a xor b = (~a | ~b) & (a | b)
  // = ~(a & b) & ~(~a & ~b)
  return ~(~x & ~y) & ~(x & y);
}

虽然这题比较简单，但我还是想了好久。主要是我把这个德摩根定律忘得差不多了，记得在离散数学里讲过。

利用的性质主要是：

  • a ^ b = (~a | ~b) & (a | b)
  • (~a | ~b) = ~(a & b)
